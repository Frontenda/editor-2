    mode: "coffee"
    text: ""

    reset Sets the content of the editor to the given content and also resets any cursor position or selection.

      reset = (content="") ->
    editor.setValue(content)
    editor.moveCursorTo(0, 0)
    editor.session.selection.clearSelection()

  reset(I.text)
